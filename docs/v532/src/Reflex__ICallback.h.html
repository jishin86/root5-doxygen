namespace Reflex {
// forward declarations
class Type;
class Member;

/**
 * @class Callback Callback.h Reflex/Callback.h
 * @author Pere Mato
 * @date 12/11/2004
 * @ingroup Ref
 */
class RFLX_API ICallback {
public:
   /** constructor */
   ICallback() {}

   /** destructor */
   virtual ~ICallback() {}

   /**
    * operator call (virtual)
    */
   virtual void operator ()(const Type&) = 0;
   virtual void operator ()(const Member&) = 0;

};    // class ICallback

} // namespace Reflex
